# Team Mode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement a PM-centered multi-agent team runtime where up to 10 workers collaborate under a project manager, with real-time observability, mailbox communication, and main session intervention.

**Architecture:** TeamRuntime is a new class that orchestrates a PM agent loop and member subagents. It extends BackgroundTaskManager with `team` type, adds mailbox injection to runSubSession, and exposes unified background tools (status/events/send). UI gets a TeamDetailPanel with per-member real-time tool timeline.

**Tech Stack:** TypeScript, Vitest, Zustand, React, Electron IPC (existing stack)

// packages/core/src/team/team-mailbox.ts

export class RingBuffer<T> {
  private buffer: T[];
  private head = 0;
  private count = 0;
  private readonly capacity: number;

  constructor(capacity: number) {
    this.capacity = capacity;
    this.buffer = new Array(capacity);
  }

  push(item: T): void {
    const index = (this.head + this.count) % this.capacity;
    if (this.count === this.capacity) {
      // Overwrite oldest
      this.buffer[this.head] = item;
      this.head = (this.head + 1) % this.capacity;
    } else {
      this.buffer[index] = item;
      this.count++;
    }
  }

  getAll(): T[] {
    const result: T[] = [];
    for (let i = 0; i < this.count; i++) {
      result.push(this.buffer[(this.head + i) % this.capacity]);
    }
    return result;
  }

  tail(n: number): T[] {
    const all = this.getAll();
    if (n >= all.length) return all;
    return all.slice(all.length - n);
  }

  get size(): number {
    return this.count;
  }
}

export class Mailbox<T> {
  private messages: T[] = [];

  push(msg: T): void {
    this.messages.push(msg);
  }

  drain(): T[] {
    const result = [...this.messages];
    this.messages = [];
    return result;
  }

  peek(): T[] {
    return [...this.messages];
  }

  isEmpty(): boolean {
    return this.messages.length === 0;
  }

  get size(): number {
    return this.messages.length;
  }
}

import { Mailbox, RingBuffer } from './team/team-mailbox.js';
import type { TeamMessage, TeamEvent } from './team/team-types.js';

// Add 'team' to TaskType union:
export type TaskType = 'sub-session' | 'background' | 'team';

// Add to BackgroundTaskManager class:
private teamMailboxes = new Map<string, Mailbox<TeamMessage>>();
private teamEvents = new Map<string, RingBuffer<TeamEvent>>();

registerTeam(teamId: string, description: string): string {
  const id = this.register({
    id: teamId,
    type: 'team',
    description,
    status: 'running',
  });
  this.teamMailboxes.set(id, new Mailbox<TeamMessage>());
  this.teamEvents.set(id, new RingBuffer<TeamEvent>(500));
  return id;
}

completeTeam(id: string, result: string): void {
  this.update(id, { status: 'completed', result });
}

failTeam(id: string, error: string): void {
  this.update(id, { status: 'failed', error });
}

sendMessage(id: string, message: TeamMessage): void {
  const mailbox = this.teamMailboxes.get(id);
  if (!mailbox) throw new Error(`No team mailbox for ${id}`);
  mailbox.push(message);
}

getMailbox(id: string): TeamMessage[] {
  const mailbox = this.teamMailboxes.get(id);
  if (!mailbox) return [];
  return mailbox.peek();
}

drainMailbox(id: string): TeamMessage[] {
  const mailbox = this.teamMailboxes.get(id);
  if (!mailbox) return [];
  return mailbox.drain();
}

emitEvent(id: string, event: TeamEvent): void {
  const buffer = this.teamEvents.get(id);
  if (!buffer) throw new Error(`No team event buffer for ${id}`);
  buffer.push(event);
}

getEvents(id: string): TeamEvent[] {
  const buffer = this.teamEvents.get(id);
  if (!buffer) return [];
  return buffer.getAll();
}

// packages/core/src/team/team-concurrency.ts
import type { AgentType, TeamConcurrencyPolicy } from './team-types.js';

export const READ_ONLY_TYPES: AgentType[] = ['explore', 'plan', 'security-auditor'];
export const WRITE_TYPES: AgentType[] = ['general', 'refactor', 'frontend-designer'];

interface RunningMember {
  memberId: string;
  agentType: AgentType;
}

export class TeamConcurrencyController {
  private running: Map<string, AgentType> = new Map();
  private policy: TeamConcurrencyPolicy;

  constructor(policy: TeamConcurrencyPolicy) {
    this.policy = policy;
  }

  canStart(agentType: AgentType): boolean {
    // Check total concurrent limit
    if (this.running.size >= this.policy.maxConcurrent) {
      return false;
    }

    // Check per-type limit
    const typeCount = [...this.running.values()].filter((t) => t === agentType).length;
    if (typeCount >= this.policy.maxPerType) {
      return false;
    }

    // Check write concurrency limit
    if (WRITE_TYPES.includes(agentType)) {
      const writeCount = [...this.running.values()].filter((t) => WRITE_TYPES.includes(t)).length;
      if (writeCount >= this.policy.maxWriteConcurrent) {
        return false;
      }
    }

    // Check read concurrency limit
    if (READ_ONLY_TYPES.includes(agentType)) {
      const readCount = [...this.running.values()].filter((t) => READ_ONLY_TYPES.includes(t)).length;
      if (readCount >= this.policy.maxReadConcurrent) {
        return false;
      }
    }

    return true;
  }

  start(memberId: string, agentType: AgentType): void {
    this.running.set(memberId, agentType);
  }

  complete(memberId: string): void {
    this.running.delete(memberId);
  }

  getRunning(): RunningMember[] {
    return [...this.running.entries()].map(([memberId, agentType]) => ({
      memberId,
      agentType,
    }));
  }
}

// packages/core/src/team/team-member.ts
import { runSubSession } from '../sub-session.js';
import { Mailbox } from './team-mailbox.js';
import type {
  AgentType,
  TeamMessage,
  TeamEvent,
  TeamMemberState,
} from './team-types.js';
import { TeamEventType, TeamMemberStatus } from './team-types.js';

export interface TeamMemberOptions {
  id: string;
  name: string;
  agentType: AgentType;
  taskId: string;
  taskDescription: string;
  onEvent: (event: TeamEvent) => void;
  onProgress: (memberId: string, progress: string) => void;
}

export class TeamMember {
  readonly id: string;
  readonly name: string;
  readonly agentType: AgentType;
  private taskId: string;
  private taskDescription: string;
  private mailbox: Mailbox<TeamMessage>;
  private _status: TeamMemberStatus = TeamMemberStatus.Idle;
  private onEvent: (event: TeamEvent) => void;
  private onProgress: (memberId: string, progress: string) => void;
  private abortController: AbortController;
  private startedAt?: number;
  private completedAt?: number;
  private result?: string;

  constructor(options: TeamMemberOptions) {
    this.id = options.id;
    this.name = options.name;
    this.agentType = options.agentType;
    this.taskId = options.taskId;
    this.taskDescription = options.taskDescription;
    this.mailbox = new Mailbox<TeamMessage>();
    this.onEvent = options.onEvent;
    this.onProgress = options.onProgress;
    this.abortController = new AbortController();
  }

  get status(): TeamMemberStatus {
    return this._status;
  }

  get pendingMessages(): number {
    return this.mailbox.size;
  }

  async start(): Promise<void> {
    this._status = TeamMemberStatus.Running;
    this.startedAt = Date.now();

    this.emitEvent(TeamEventType.MemberStarted, {
      memberId: this.id,
      agentType: this.agentType,
      taskId: this.taskId,
    });

    try {
      const result = await runSubSession({
        task: this.taskDescription,
        agentType: this.agentType,
        mailbox: this.mailbox,
        signal: this.abortController.signal,
        onProgress: (progress: string) => {
          this.onProgress(this.id, progress);
        },
      });

      this._status = TeamMemberStatus.Completed;
      this.completedAt = Date.now();
      this.result = result.result;

      this.emitEvent(TeamEventType.MemberCompleted, {
        memberId: this.id,
        taskId: this.taskId,
        result: result.result,
        toolCalls: result.toolCalls,
        turns: result.turns,
      });
    } catch (error) {
      this._status = TeamMemberStatus.Failed;
      this.completedAt = Date.now();

      this.emitEvent(TeamEventType.MemberFailed, {
        memberId: this.id,
        taskId: this.taskId,
        reason: error instanceof Error ? error.message : String(error),
      });
    }
  }

  sendMessage(msg: TeamMessage): void {
    this.mailbox.push(msg);
  }

  abort(reason: string): void {
    this._status = TeamMemberStatus.Failed;
    this.abortController.abort(reason);
    this.completedAt = Date.now();

    this.emitEvent(TeamEventType.MemberFailed, {
      memberId: this.id,
      taskId: this.taskId,
      reason,
    });
  }

  getState(): TeamMemberState {
    return {
      id: this.id,
      agentType: this.agentType,
      name: this.name,
      status: this._status,
      taskId: this.taskId,
      progress: undefined,
      startedAt: this.startedAt,
      completedAt: this.completedAt,
    };
  }

  private emitEvent(type: TeamEventType, payload: Record<string, unknown>): void {
    this.onEvent({
      id: `${this.id}-${type}-${Date.now()}`,
      type,
      payload,
      timestamp: Date.now(),
    });
  }
}

import { randomUUID } from 'crypto';
import type {
  TeamTask,
  TeamEvent,
  TeamManagerState,
  TeamMemberState,
  TeamSharedContext,
  TeamConcurrencyPolicy,
} from './team-types.js';
import {
  TeamTaskStatus,
  TeamEventType,
  DEFAULT_CONCURRENCY_POLICY,
} from './team-types.js';

export interface TeamManagerOptions {
  teamId: string;
  objective: string;
  onEvent: (event: TeamEvent) => void;
  onMemberRequest: (action: string, payload: Record<string, unknown>) => void;
  concurrencyPolicy?: TeamConcurrencyPolicy;
}

interface CreateTaskInput {
  title: string;
  description: string;
  dependencies: string[];
}

export class TeamManager {
  private teamId: string;
  private objective: string;
  private tasks: Map<string, TeamTask> = new Map();
  private members: TeamMemberState[] = [];
  private sharedContext: TeamSharedContext;
  private synthesis?: string;
  private status: 'planning' | 'running' | 'synthesizing' | 'completed' | 'failed' = 'planning';
  private onEvent: (event: TeamEvent) => void;
  private onMemberRequest: (action: string, payload: Record<string, unknown>) => void;
  private createdAt: number;
  private updatedAt: number;

  constructor(options: TeamManagerOptions) {
    this.teamId = options.teamId;
    this.objective = options.objective;
    this.onEvent = options.onEvent;
    this.onMemberRequest = options.onMemberRequest;
    this.createdAt = Date.now();
    this.updatedAt = Date.now();
    this.sharedContext = {
      objective: options.objective,
      constraints: [],
      files: [],
      decisions: [],
    };
  }

  createTask(input: CreateTaskInput): TeamTask {
    const task: TeamTask = {
      id: randomUUID(),
      title: input.title,
      description: input.description,
      status: TeamTaskStatus.Pending,
      dependencies: input.dependencies,
      createdAt: Date.now(),
      updatedAt: Date.now(),
    };
    this.tasks.set(task.id, task);
    this.updatedAt = Date.now();
    return task;
  }

  assignTask(taskId: string, memberId: string): void {
    const task = this.tasks.get(taskId);
    if (!task) throw new Error(`Task ${taskId} not found`);
    task.assignee = memberId;
    task.status = TeamTaskStatus.InProgress;
    task.updatedAt = Date.now();
    this.updatedAt = Date.now();
    this.status = 'running';

    this.emitEvent(TeamEventType.TaskAssigned, {
      taskId,
      memberId,
      title: task.title,
    });
  }

  cancelTask(taskId: string): void {
    const task = this.tasks.get(taskId);
    if (!task) throw new Error(`Task ${taskId} not found`);
    task.status = TeamTaskStatus.Cancelled;
    task.updatedAt = Date.now();
    this.updatedAt = Date.now();
  }

  getTask(taskId: string): TeamTask | undefined {
    return this.tasks.get(taskId);
  }

  sendMemberMessage(memberId: string, content: string): void {
    this.onMemberRequest('send_message', {
      from: 'pm',
      to: memberId,
      content,
    });

    this.emitEvent(TeamEventType.MessageSent, {
      from: 'pm',
      to: memberId,
      content,
    });
  }

  broadcast(content: string): void {
    this.onMemberRequest('broadcast', { content });

    this.emitEvent(TeamEventType.MessageSent, {
      from: 'pm',
      to: 'all',
      content,
    });
  }

  addConstraint(constraint: string): void {
    this.sharedContext.constraints.push(constraint);
    this.updatedAt = Date.now();

    this.emitEvent(TeamEventType.ConstraintAdded, { constraint });
  }

  submitSynthesis(synthesis: string): void {
    this.synthesis = synthesis;
    this.status = 'synthesizing';
    this.updatedAt = Date.now();

    this.emitEvent(TeamEventType.SynthesisReady, { synthesis });
  }

  handleTaskCompleted(taskId: string, result: string): void {
    const task = this.tasks.get(taskId);
    if (!task) return;
    task.status = TeamTaskStatus.Completed;
    task.result = result;
    task.updatedAt = Date.now();
    this.updatedAt = Date.now();

    this.emitEvent(TeamEventType.TaskCompleted, { taskId, result });
  }

  handleTaskFailed(taskId: string, reason: string): void {
    const task = this.tasks.get(taskId);
    if (!task) return;
    task.status = TeamTaskStatus.Failed;
    task.updatedAt = Date.now();
    this.updatedAt = Date.now();

    this.emitEvent(TeamEventType.TaskFailed, { taskId, reason });
  }

  getSharedContext(): TeamSharedContext {
    return { ...this.sharedContext };
  }

  getState(): TeamManagerState {
    return {
      teamId: this.teamId,
      objective: this.objective,
      members: [...this.members],
      tasks: [...this.tasks.values()],
      sharedContext: { ...this.sharedContext },
      concurrencyPolicy: DEFAULT_CONCURRENCY_POLICY,
      events: [],
      synthesis: this.synthesis,
      status: this.status,

// packages/core/src/team/team-runtime.ts
import { randomUUID } from 'crypto';
import { TeamManager } from './team-manager.js';
import { TeamMember } from './team-member.js';
import { TeamConcurrencyController } from './team-concurrency.js';
import { RingBuffer } from './team-mailbox.js';
import type {
  TeamRuntime,
  TeamMessage,
  TeamEvent,
  TeamManagerState,
  TeamMemberState,
  TeamConcurrencyPolicy,
  AgentType,
} from './team-types.js';
import {
  DEFAULT_CONCURRENCY_POLICY,
  TeamEventType,
  TeamMemberStatus,
} from './team-types.js';

export interface TeamRuntimeOptions {
  teamId: string;
  objective: string;
  members: Array<{ id: string; name: string; agentType: AgentType }>;
  concurrencyPolicy?: TeamConcurrencyPolicy;
}

export class TeamRuntimeImpl implements TeamRuntime {
  private teamId: string;
  private objective: string;
  private memberConfigs: Array<{ id: string; name: string; agentType: AgentType }>;
  private members: Map<string, TeamMember> = new Map();
  private manager: TeamManager;
  private concurrency: TeamConcurrencyController;
  private eventLog: RingBuffer<TeamEvent>;
  private status: 'planning' | 'running' | 'synthesizing' | 'completed' | 'failed' = 'planning';
  private stopped = false;

  constructor(options: TeamRuntimeOptions) {
    this.teamId = options.teamId;
    this.objective = options.objective;
    this.memberConfigs = options.members;
    this.eventLog = new RingBuffer<TeamEvent>(500);
    this.concurrency = new TeamConcurrencyController(
      options.concurrencyPolicy ?? DEFAULT_CONCURRENCY_POLICY
    );

    this.manager = new TeamManager({
      teamId: this.teamId,
      objective: this.objective,
      onEvent: (event) => this.handleEvent(event),
      onMemberRequest: (action, payload) => this.handleMemberRequest(action, payload),
    });
  }

  async start(): Promise<void> {
    // Initialize members
    for (const config of this.memberConfigs) {
      const member = new TeamMember({
        id: config.id,
        name: config.name,
        agentType: config.agentType,
        taskId: '',
        taskDescription: this.objective,
        onEvent: (event) => this.handleEvent(event),
        onProgress: (memberId, progress) => this.handleProgress(memberId, progress),
      });
      this.members.set(config.id, member);
    }
  }

  sendMessage(msg: TeamMessage): void {
    this.handleEvent({
      id: randomUUID(),
      type: TeamEventType.MessageReceived,
      payload: { from: msg.from, to: msg.to, content: msg.content },
      timestamp: Date.now(),
    });

    if (msg.to === 'pm') {
      // Route to PM manager
      return;
    }

    if (msg.to === 'all') {
      // Broadcast to all members
      for (const member of this.members.values()) {
        member.sendMessage(msg);
      }
      return;
    }

    // Route to specific member
    const member = this.members.get(msg.to);
    if (member) {
      member.sendMessage(msg);
    }
  }

  getStatus(): TeamManagerState {
    const memberStates: TeamMemberState[] = [...this.members.values()].map((m) => m.getState());
    const managerState = this.manager.getState();
    return {
      ...managerState,
      members: memberStates,
      status: this.status,
      events: this.eventLog.getAll(),
    };
  }

  getEvents(since?: number): TeamEvent[] {
    const all = this.eventLog.getAll();
    if (since === undefined) return all;
    return all.filter((e) => e.timestamp > since);
  }

  async stop(): Promise<void> {
    this.stopped = true;
    for (const member of this.members.values()) {
      if (member.status === TeamMemberStatus.Running) {
        member.abort('Team stopped');
      }
    }
    this.status = 'completed';
  }

  private handleEvent(event: TeamEvent): void {
    this.eventLog.push(event);
  }

  private handleMemberRequest(action: string, payload: Record<string, unknown>): void {
    if (action === 'send_message') {
      const memberId = payload.to as string;
      const member = this.members.get(memberId);
      if (member) {
        member.sendMessage({
          id: randomUUID(),
          from: payload.from as string,
          to: memberId,
          content: payload.content as string,
          timestamp: Date.now(),
        });
      }
    } else if (action === 'broadcast') {
      for (const member of this.members.values()) {
        member.sendMessage({
          id: randomUUID(),
          from: 'pm',
          to: member.id,
          content: payload.content as string,
          timestamp: Date.now(),
        });
      }
    }
  }

  private handleProgress(memberId: string, progress: string): void {
    // Could emit progress events or update member state
  }
}

export const backgroundEventsTool: ToolDefinition = {
  name: 'background_events',
  description: 'Get recent events from a background team. Use to monitor activity and progress.',
  parameters: {
    type: 'object',
    properties: {
      taskId: {
        type: 'string',
        description: 'The team/task ID',
      },

        type: 'number',
        description: 'Max number of recent events to return (default: 20)',
      },

        type: 'number',
        description: 'Only return events after this timestamp',
      },
    },
    required: ['taskId'],
  },
  async execute(params: any, context: any): Promise<string> {
    const limit = params.limit ?? 20;
    let events = context.backgroundTasks.getEvents(params.taskId);

    if (params.since) {
      events = events.filter((e: any) => e.timestamp > params.since);
    }

    // Take last N events
    events = events.slice(-limit);

    if (events.length === 0) {
      return 'No events found.';
    }

    return events.map((e: any) =>
      `[${new Date(e.timestamp).toISOString()}] ${e.type}: ${JSON.stringify(e.payload)}`
    ).join('\n');
  },
};

// packages/ui/src/stores/team-store.ts
import { create } from 'zustand';
import type { TeamManagerState, TeamEvent } from '@jdcagnet/core';

interface TeamStoreState {
  teams: Map<string, TeamManagerState>;
  events: Map<string, TeamEvent[]>;
  activeTeamId: string | null;
  expandedMembers: Set<string>;

  setTeamStatus: (teamId: string, status: TeamManagerState) => void;
  addEvent: (teamId: string, event: TeamEvent) => void;
  setActiveTeam: (teamId: string | null) => void;
  toggleMemberExpanded: (memberId: string) => void;
  reset: () => void;
}

const MAX_EVENTS_PER_TEAM = 200;

export const useTeamStore = create<TeamStoreState>((set, get) => ({
  teams: new Map(),
  events: new Map(),
  activeTeamId: null,
  expandedMembers: new Set(),

  setTeamStatus: (teamId, status) => {
    set((state) => {
      const teams = new Map(state.teams);
      teams.set(teamId, status);
      return { teams };
    });
  },

  addEvent: (teamId, event) => {
    set((state) => {
      const events = new Map(state.events);
      const teamEvents = [...(events.get(teamId) ?? []), event];
      // Cap at MAX_EVENTS_PER_TEAM
      if (teamEvents.length > MAX_EVENTS_PER_TEAM) {
        teamEvents.splice(0, teamEvents.length - MAX_EVENTS_PER_TEAM);
      }
      events.set(teamId, teamEvents);
      return { events };
    });
  },

  setActiveTeam: (teamId) => {
    set({ activeTeamId: teamId });
  },

  toggleMemberExpanded: (memberId) => {
    set((state) => {
      const expanded = new Set(state.expandedMembers);
      if (expanded.has(memberId)) {
        expanded.delete(memberId);
      } else {
        expanded.add(memberId);
      }
      return { expandedMembers: expanded };
    });
  },

  reset: () => {
    set({
      teams: new Map(),
      events: new Map(),
      activeTeamId: null,
      expandedMembers: new Set(),
    });
  },
}));

// packages/ui/src/hooks/useTeamEvents.ts
import { useEffect } from 'react';
import { useTeamStore } from '../stores/team-store.js';

interface IpcClient {
  on: (channel: string, callback: (...args: any[]) => void) => () => void;
  invoke: (channel: string, ...args: any[]) => Promise<any>;
}

export function useTeamEvents(ipc: IpcClient, teamId: string | null) {
  const addEvent = useTeamStore((s) => s.addEvent);
  const setTeamStatus = useTeamStore((s) => s.setTeamStatus);

  useEffect(() => {
    if (!teamId) return;

    const unsubEvent = ipc.on('team:event', (event: any) => {
      if (event.teamId === teamId || event.payload?.teamId === teamId) {
        addEvent(teamId, event);
      }
    });

    const unsubStatus = ipc.on('team:status', (status: any) => {
      if (status.teamId === teamId) {
        setTeamStatus(teamId, status);
      }
    });

    // Initial fetch
    ipc.invoke('team:getStatus', teamId).then((status) => {
      if (status) setTeamStatus(teamId, status);
    });

    return () => {
      unsubEvent();
      unsubStatus();
    };
  }, [teamId, ipc, addEvent, setTeamStatus]);
}
